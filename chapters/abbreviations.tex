\addcontentsline{toc}{chapter}{СПИСОК ИСПОЛЬЗУЕМЫХ СОКРАЩЕНИЙ}
\begin{enumerate}
	\item Оптоволокно -- оптическое волокно
	\item SMF (single mode fiber) -- одномодовое оптоволокно
	\item MMF (multiple mode fiber) -- многомодовое оптоволокно
	\item NLSE (nonlinear Schrödinger equation) -- нелинейное уравнение Шрёдингера
	\item CD (chromatic dispersion) -- хроматическая дисперсия
	\item GVD (group velocity dispersion) -- дисперсия групповой скорости
	\item SSFM (split-step Fourier method) -- метод расщепления Фурье
	\item DBP (digital backpropogation) -- алгоритм цифрового обратного распространения
	\item PBM (perturbation-based method) -- метод компенсации искажений сигнала на основе теории возмущений
	\item NN (neural network) -- нейронная сеть
	\item SGD (stochastic gradient descent) -- стохастический градиентный спуск
\end{enumerate}
